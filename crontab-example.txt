# مثال لملف crontab لإلغاء الطلبات تلقائياً
# تشغيل كل 5 دقائق

# إضافة المتغيرات البيئية
CRON_API_KEY=your-secret-key-here
API_URL=https://yourdomain.com

# تشغيل script الإلغاء التلقائي كل 5 دقائق
*/5 * * * * cd /path/to/your/project && node scripts/auto-cancel-orders.js >> logs/cron.log 2>&1

# أو استخدام curl مباشرة
*/5 * * * * curl -H "x-api-key: your-secret-key-here" https://yourdomain.com/api/cron/auto-cancel-orders >> logs/cron.log 2>&1

# تشغيل كل دقيقة (للاختبار)
# * * * * * cd /path/to/your/project && node scripts/auto-cancel-orders.js >> logs/cron.log 2>&1

# تشغيل كل ساعة
# 0 * * * * cd /path/to/your/project && node scripts/auto-cancel-orders.js >> logs/cron.log 2>&1

# ملاحظات:
# 1. استبدل 'your-secret-key-here' بمفتاح API حقيقي
# 2. استبدل 'yourdomain.com' برابط موقعك
# 3. استبدل '/path/to/your/project' بالمسار الفعلي لمشروعك
# 4. تأكد من وجود مجلد 'logs' أو قم بإنشائه
# 5. يمكنك تعديل التوقيت حسب احتياجاتك
